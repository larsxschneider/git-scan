898 verify_filename(prefix, arg, 1);
899 }
900 if (verify) {
901 if (revs_count == 1) {
902 show_rev(type, sha1, name);
903 return 0;
904 } else if (revs_count == 0 && show_default())

906 die_no_single_rev(quiet);
