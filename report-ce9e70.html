534static int match_extension(const char *filename, const char *ext)
535{
536 int prefixlen = strlen(filename) - strlen(ext);
537
538 /*
539 * We need 1 character for the '.', and 1 character to ensure that the
540 * prefix is non-empty (k.e., we don't match .tar.gz with no actual
541 * filename).
542 */
543 if (prefixlen < 2 || filename[prefixlen - 1] != '.')
544 return 0;
545 return !strcmp(filename + prefixlen, ext);
546}
547
548const char *archive_format_from_filename(const char *filename)
549{
550 int i;
551
552 for (i = 0; i < nr_archivers; i++)
553 if (match_extension(filename, archivers[i]->name))
554 return archivers[i]->name;
555 return NULL((void*)0);
556}
